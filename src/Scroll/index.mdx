# Scroll Component

基于 iscroll 封装的 滚动组件

<Playground>
  {()=>{
    const Temp = () => {
      const [scrollToY, setScrollToY] = React.useState(-100);
      const [endPullUp, setEndPullUp] = React.useState(false);
      React.useEffect(()=>{
        setTimeout(() => {
          setScrollToY('null');
        });
      }, [scrollToY])

      return (
          <div style={{'position': 'relative', 'height':'100px'}}>
              <Scroll
                  showPullDown={true}
                  showPullUp={true}
                  endPullUp={endPullUp}
                  bottom={20}
                  top={30}
                  scrollToY={scrollToY}
                  scrollPullDown={(pullDownComplete)=>{
                      setTimeout(()=>{
                          pullDownComplete(()=>{
                              console.log('ok')
                          })
                      },3000);
                  }}
                  scrollPullUp={(pullUpComplete)=>{
                      setTimeout(()=>{
                          pullUpComplete(()=>{
                              console.log('ok')
                          })
                      },3000);
                  }}
                  onScrollY={(y)=>{
                    console.log(y)
                  }}
              >
                  <div style={{'width':'100%'}}>
                    <a onClick={()=>{
                      setScrollToY(-100)
                      }}>try pulling me 1</a><br/>
                    try pulling me 2<br/>
                    try pulling me 3<br/>
                    try pulling me 4<br/>
                    try pulling me 5<br/>
                    try pulling me 6<br/>
                    try pulling me 7<br/>
                    try pulling me 8<br/>
                    try pulling me 9<br/>
                    try pulling me 10<br/>
                    try pulling me 11<br/>
                    <a onClick={()=>{
                      setEndPullUp(true)
                      }}>try pulling me 12</a><br/>
                  </div>
              </Scroll>
          </div>
      );
    };

    return <Temp/>;
  }}
</Playground>

|      参数      |                                    说明                                    |   类型   | 默认值 |
| -------------- | -------------------------------------------------------------------------- | -------- | ------ |
| showPullDown   | 是否显示下拉                                                               | boolean  | false  |
| showPullUp     | 是否显示上拉                                                               | boolean  | false  |
| endPullUp      | 上拉是否结尾                                                               | boolean  | false  |
| scrollPullDown | 下拉回调方法                                                               | function |        |
| scrollPullUp   | 上拉回调方法                                                               | function |        |
| onScrollY      | 上下滚动事件                                                               | function |        |
| scrollToY      | 滚动到y轴位置（'null'为不滚动，请注意state变化，需在生命周期设置为'null'） | number   |        |
