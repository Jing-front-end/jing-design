# Input 输入框

loading 等 UI

## 一般用法

<Playground>
  {() => {
    const Temp = () => {
      const [inputValue, setIsInputValue] = React.useState('');
      const [inputValue1, setIsInputValue1] = React.useState('');
      return (
        <div style={{ display: 'flex', flexDirection: 'column'}}>
            <span>普通模式</span>
            <Input
                style={{ marginBottom: '15px' }}
                value={inputValue}
                onChange={value => {
                    {/* console.log('value', value); */}
                    setIsInputValue(value);
                }}
                />
            <span>错误框</span>
            <Input
                style={{ marginBottom: '15px' }}
                error
                value={inputValue1}
                onChange={value => {
                    {/* console.log('value', value); */}
                    setIsInputValue1(value);
                }}
            />
            <span>禁止输入</span>
            <Input
                style={{ marginBottom: '15px' }}
                disabled
            />
        </div>
      );
    };
    return <Temp />;
  }}
</Playground>
