import { Playground } from 'docz';
import KeyBoard from './index.tsx';
import Toast from '../Toast';

# KeyBoard 键盘

## 使用 KeyBoardNumber
<Playground>
  {
    () => {
      const KeyBoardExample = () => {

        return (
          <div>
              <KeyBoard.Number 
                onHandleValue={(value) => {
                  Toast.show(value)
                }}
                onHandleDelete={() => {
                  Toast.show('刪除')
                }}/>
          </div>
        )
      }
      return <KeyBoardExample />
    }
  }
</Playground>
