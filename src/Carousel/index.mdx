## 组件写法
<Playground>
  {
    () => {
      const CarouselExample = () => {
        const data = [
          {
            id: 1,
            img: 'https://pic.jingqb.com/pic/banner/2019/PIC000000274.jpg',
            title: '我是第一'
          },
          {
            id: 2,
            img: 'https://pic.jingqb.com/pic/banner/2019/PIC000000264.jpg',
            title: '我是第二'
          },
          {
            id: 3,
            img: 'https://pic.jingqb.com/pic/banner/2019/PIC000000217.png',
            title: '我是第三'
          },
          {
            id: 4,
            img: 'https://pic.jingqb.com/pic/banner/2019/PIC000000268.png',
            title: '我是第四'
          },
          {
            id: 5,
            img: 'https://pic.jingqb.com/pic/banner/2017/PIC000000101.png',
            title: '我是第五'
          },
        ]
        const [swiper, setSwiper] = React.useState(null);
          const goNext = () => {
            if (swiper !== null) {
              swiper.slideNext();
            }
          };
        
          const goPrev = () => {
            if (swiper !== null) {
              swiper.slidePrev();
            }
          };
        return (
          <div>
            <Carousel getSwiper={setSwiper}>
              {
                data.map((item) => (
                  <div key={item.id}>
                    <img style={{ height: '10rem'}} src={item.img} alt={item.title} />
                    <span style={{ position: 'absolute', top: '20%', left: '50%', color: 'yellow' }} > { item.title }</span>
                  </div>
                ))
              }
            </Carousel>
            <button onClick={goPrev}>Prev</button>
            <button onClick={goNext}>Next</button>
          </div>
        )
      }
      return <CarouselExample />
    }
  }
</Playground>
