# DetailList 详情列表

## 使用
<Playground>
  {
    () => {
      const DetailListExample = () => {
        const data = [
          {
            id: 1,
            leftText: "订单编号",
            orderid: "20200107101217452271",
            payDate: "2020-01-07 10:12:19",
            productId: "ZH00VM",
            productName: "天玑聚富",
            tradeMoney: 10000,
          },
        ]
        return (
          <div>
            {
              data.map((item) => (
                <DetailList 
                  leftText={item.leftText}
                  rightChildren={item.orderid}
                />
              ))
            }
          </div>
        )
      }
      return <DetailListExample />
    }
  }
</Playground>
