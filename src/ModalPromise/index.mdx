# ModalPromise 模态框

## Confirm用法
<Playground>
  {()=>{
    const Temp = () => {
      const [isShow, setIsShow] = React.useState(false);

      return (
          <div>
            <Button.Group>
              <Button.Self color="blue" onClick={()=>{
                  Modal.Confirm().then((res) => {
                    console.log(res);
                  }, (res) => {
                      console.log(res);
                  })
                }} primary>确认</Button.Self>
            </Button.Group>
          </div>
      );
    };

    return <Temp/>;
  }}
</Playground>

## 自定义模态框
<Playground>
  {()=>{
    const Temp = () => {
      const [isShow, setIsShow] = React.useState(false);

      function Demo(props) {
          return (
              <div>
                  <Modal.Header {...props}>测试</Modal.Header>
                  <Modal.Body>
                      <div>测试</div>
                      <div>测试</div>
                      <div>测试</div>
                  </Modal.Body>
                  <Modal.Footer></Modal.Footer>
              </div>
          )
      }

      return (
          <div>
            <Button.Group>
              <Button.Self color="blue" onClick={()=>{
                  Modal.Dialog(Demo).then((res) => {
                    console.log(res);
                  }, (res) => {
                      console.log(res);
                  })
                }} primary>自定义模态框</Button.Self>
            </Button.Group>
          </div>
      );
    };

    return <Temp/>;
  }}
</Playground>
